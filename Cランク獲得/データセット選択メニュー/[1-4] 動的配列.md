# 動的配列

https://paiza.jp/works/mondai/data_structure/data_structure__array_boss

<details><summary>C++ 解答</summary><br>

```cpp
#include <iostream>
#include <vector>
using namespace std;

int main() {
    int N, Q, query, x;
    cin >> N >> Q;
    vector<int> A(N);
    for (int i = 0; i < N; i++) {
        cin >> A.at(i);
    }
    
    for (int i = 0; i < Q; i++) {
        cin >> query;
        if (query == 0) {
            cin >> x;
            A.push_back(x);
        } else if (query == 1) {
            A.pop_back();
        } else if (query == 2) {
            for (int j = 0; j < A.size() - 1; j++) {
                cout << A.at(j) << " ";
            }
            cout << A.back() << endl;
        }
    }
}
```

別解

```cpp
#include <iostream>
using namespace std;

int main() {
    int N, Q, query, x;
    cin >> N >> Q;
    int A[200];
    for (int i = 0; i < N; i++) {
        cin >> A[i];
    }
    
    int size = N;
    
    for (int i = 0; i < Q; i++) {
        cin >> query;
        if (query == 0) {
            cin >> x;
            A[size] = x;
            size++;
        } else if (query == 1) {
            size--;
        } else if (query == 2) {
            for (int j = 0; j < size - 1; j++) {
                cout << A[j] << " ";
            }
            cout << A[size - 1] << endl;
        }
    }
}
```

            cin >> x;

            size++;

            size--;
        } else if (query == 2) {
            for (int j = 0; j < size - 1; j++) {
                cout << arr[j] << " ";
            }

    }
}
