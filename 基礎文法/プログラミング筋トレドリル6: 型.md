https://paiza.jp/works/mondai/cgc_problems_006/cgc_problems__pr13_lv06

string → int

```cpp
#include <iostream>
#include <string>
using namespace std;
// ----------------------------------------
// [型 - 型変換1]
// ----------------------------------------
// 変数aをint型(整数型)に変換してください

int main(void) {
	string s = "813";
	int a = atoi(s.c_str()); // atoiを使用する
	
	cout << a << endl;
}
```

ChatGPTに言われたことをまとめると…<br>
`atoi` はC言語由来の関数<br>
名前の意味は ASCII to Integer<br>
`atoi` が受け取れるのは `char*`（C文字列） だけ<br>
そのため、`c_str()` で `string` をC文字列（`const char*`）に変換している

別解

```cpp
#include <iostream>
#include <string>
using namespace std;

int main() {
	string s = "813";
	int a = stoi(s);
	
	cout << a << endl;
}
```

別解1

```cpp
#include <iostream>
#include <format>
using namespace std;

int main() {
	string a = format("{}", 11813);
	
	cout << a << endl;
}
```

別解2<br>
`ss >> a;` は `getline(ss, a);` や `a = ss.str();` でも可。

```cpp
#include <iostream>
#include <sstream>
using namespace std;

int main() {
    int n = 11813;
    string a;

    stringstream ss;
    ss << n;
    ss >> a;

    cout << a << endl;
}
```

別解1

```cpp
#include <iostream>
#include <string>
using namespace std;

int main() {
	string s = "1.11111";
	double a = stod(s);
	
	cout << a << endl;
```

```cpp
#include <iostream>
#include <sstream>
#include <string>
using namespace std;

int main() {
	string s = "1.11111";
    stringstream ss(s);

	double a;
    ss >> a;
	
	cout << a << endl;
```

```cpp
#include <iostream>
#include <cstdlib>
using namespace std;

int main() {
    const char* s = "1.11111";
    char* end;

    double a = strtod(s, &end);

    cout << a << endl;
}
```

<details><summary>型 - 型変換 6 C++ 解答</summary><br>

https://paiza.jp/works/mondai/cgc_problems_006/cgc_problems__pr18_lv06

int → double

`double a = (double)7;` は、単に `double a = 7;` としたり、`double a = 7 / 1.0;` としても可。

```cpp
#include <iostream>
#include <iomanip>
using namespace std;
// ----------------------------------------
// [型 - 型変換6]
// ----------------------------------------
// 変数aをdouble型(実数型)に変換してください

int main(void) {
	double a = (double)7; // 値の前に(double)と書く
	
	cout << fixed << setprecision(1) << a << endl;
}
```

別解

```cpp
#include <iostream>
#include <iomanip>
using namespace std;

int main() {
	double a = static_cast<double>(7);
	
	cout << fixed << setprecision(1) << a << endl;
}
```

<details><summary>型 - 少数点以下の切り捨て 1 C++ 解答</summary><br>

(double → int)

`int a = (int)3.14;` は、単に `int a = 3.14;` としたり、`int a = 3.14 / 1;` でも可。

```cpp
#include <iostream>
using namespace std;
// ----------------------------------------
// [型 - 少数点以下の切り捨て1]
// ----------------------------------------
// 変数aの数値の小数点以下を切り捨てて、int型のデータとして出力してください

int main(void) {
	int a = (int)3.14;
	
	cout << a << endl;
}
```

別解1

```cpp
#include <iostream>
using namespace std;

int main() {
	int a = static_cast<int>(3.14);
	
	cout << a << endl;
}
```

別解2

```cpp
#include <iostream>
#include <cmath>
using namespace std;

int main() {
	int a = floor(3.14);
	
	cout << a << endl;
}
```
